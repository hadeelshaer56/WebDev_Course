        .demo-box {
            padding: 20px;

        }

<body class="p-4">

    <h2 class="mb-4">JS + Bootstrap Selectors Demo</h2>

    <!-- Select Single Item -->
    <div class="demo-box mb-4">
        <h5>Select Single Element (getElementById)</h5>
        <p id="singleText">This text will change color.</p>
        <button id="btnColor" class="btn btn-primary btn-sm">Change Color</button>
    </div>

    <!-- Select Multiple Items -->
    <div class="demo-box mb-4">
        <h5>Select Multiple Elements (querySelectorAll)</h5>
        <div class="d-flex gap-2 mb-2">
            <span class="badge bg-secondary tag">Tag A</span>
            <span class="badge bg-secondary tag">Tag B</span>
            <span class="badge bg-secondary tag">Tag C</span>
        </div>
        <button id="btnMulti" class="btn btn-success btn-sm">Highlight All Tags</button>
    </div>

    <!-- Hide / Show -->
    <div class="demo-box mb-4">
        <h5>Hide & Show Element</h5>
        <p id="hideText">This element can be hidden or shown.</p>
        <button id="btnHide" class="btn btn-danger btn-sm">Hide</button>
        <button id="btnShow" class="btn btn-info btn-sm">Show</button>
    </div>

    <!-- Change Bootstrap Classes -->
    <div class="demo-box mb-4">
        <h5>Change Classes (Bootstrap)</h5>
        <div id="box" class="p-3 bg-light border rounded">Watch my style change</div>
        <button id="btnRounded" class="btn btn-warning btn-sm mt-2">Round Corners</button>
        <button id="btnBorder" class="btn btn-dark btn-sm mt-2">Add Border</button>
    </div>

    <!-- Input Event -->
    <div class="demo-box">
        <h5>Input Event</h5>

        <p id="liveOutput" class="fw-bold"></p>

    <!-- STEP 2: JAVASCRIPT CODE -->
    <script>
// ==============================================
// STEP 2.1: SELECT SINGLE ELEMENT + CLICK EVENT
// ==============================================

// Select the paragraph by ID
const singleText = document.getElementById("singleText");

// Select the button by ID and add click event
document.getElementById("btnColor").addEventListener("click", () => {
    // Change style dynamically
    singleText.style.color = "red";
    singleText.style.fontWeight = "bold";
});


// ==============================================
// STEP 2.2: SELECT MULTIPLE ELEMENTS (querySelectorAll)
// ==============================================

// Select all elements with class "tag"
const tags = document.querySelectorAll(".tag");

// Add event to button that will change ALL tags
document.getElementById("btnMulti").addEventListener("click", () => {
    tags.forEach(t => {
        // Remove old Bootstrap classes
        t.classList.remove("bg-secondary");
        // Add new Bootstrap classes
        t.classList.add("bg-warning", "text-dark");
    });
});


// ==============================================
// STEP 2.3: HIDE AND SHOW ELEMENT
// ==============================================

// Hide element by changing CSS display property
document.getElementById("btnHide").addEventListener("click", () => {
    document.getElementById("hideText").style.display = "none";
});

// Show element by changing CSS display property
document.getElementById("btnShow").addEventListener("click", () => {
    document.getElementById("hideText").style.display = "block";
});


// ==============================================
// STEP 2.4: CHANGE BOOTSTRAP CLASSES DYNAMICALLY
// ==============================================

// Select the box
const box = document.getElementById("box");

// Toggle rounded-pill class (more rounded corners)
document.getElementById("btnRounded").addEventListener("click", () => {
    box.classList.toggle("rounded-pill");
});

// Toggle border size and color
document.getElementById("btnBorder").addEventListener("click", () => {
    box.classList.toggle("border-3");
    box.classList.toggle("border-danger");
});


// ==============================================
// STEP 2.5: INPUT EVENT (LIVE TEXT UPDATE)
// ==============================================

// Select input and output
const textInput = document.getElementById("textInput");
const liveOutput = document.getElementById("liveOutput");

// "input" event fires on every change
textInput.addEventListener("input", () => {
    liveOutput.textContent = "You typed: " + textInput.value;
});
    </script>

</body>
</html>
